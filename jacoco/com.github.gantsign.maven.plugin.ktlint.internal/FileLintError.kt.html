package com.github.gantsign.maven.plugin.ktlint.internal

import com.github.shyiko.ktlint.core.LintError

internal class FileLintError(
    val file: String,
    val line: Int,
    val col: Int,
    val ruleId: String,
    val detail: String
) {
    constructor(file: String, lintError: LintError) : this(
        file,
        lintError.line,
        lintError.col,
        lintError.ruleId,
        lintError.detail
    )
}
